#+TITLE: Development tooling
#+LANGUAGE: en
#+PROPERTY: header-args:python :tangle lib/dev.py

This module provides reproducible development tooling for daggerlib.
There are three ways to use the tooling, kept in sync:

1. *Mode 1 — bootstrap (host tools):* Uses your local emacs, git,
   ruff to tangle the minimum files needed for mode 2 to work. This
   solves the chicken-and-egg problem: dagger functions are tangled
   Python, so you need a working tangle to get them. Run
   =./tangle-nodagger.sh= once, then never again.

2. *Mode 2 — local development (dagger):* All commands run inside
   dagger containers. =./tangle.sh=, =./run.sh=, =./test.sh= are
   thin wrappers around =dagger call dev ...=. This is what you use
   for daily work. Reproducible, no nix needed.

3. *Mode 3 — CI:* Same as mode 2. CI runs
   =./run.sh && ./tangle.sh && ./test.sh=.

The bootstrap flow:

1. =./tangle-nodagger.sh= tangles only the files needed for mode 2:
   =src/lib/dev.py=, =src/lib/main.py=, =src/lib/__init__.py=,
   =tangle.sh=, =run.sh=, =run-nodagger.sh=, =test.sh=,
   =test-nodagger.sh=.
2. =./tangle.sh= (mode 2) calls =dagger call dev tangle=. Tangles
   all org files inside the container, including rewriting the files
   from step 1. From this point, everything is reproducible.
3. Use mode 2 for all daily work.

from pathlib import Path
from typing import Annotated

import dagger
from dagger import DefaultPath, dag, function, object_type

DAGGER_VERSION = "0.20.3"
ORG_PIN = (Path(__file__).resolve().parents[2] / ".org-pin").read_text().strip()
#+end_src

* Dev class

  The =Dev= class provides all development operations as dagger
  functions. The base image is configured via [[file:../TECHNICAL.org::*Module entry point][Lib.dev_debian_image]]
  so all versions live in one place. The =container= method builds the
  image with everything needed: emacs for org-babel, git for the
  pinned org-mode clone, ruff for Python formatting, pytest for tests,
  and the dagger CLI for nested calls. Pinned org-mode is cloned
  during container build so that dagger caches it.

  =_work_container= mounts the source into =/work= and optionally
  symlinks the pinned org-mode so =tangle-nodagger.sh= skips the
  clone step. All three operations build on it:

  - =tangle= tangles all org files. =tangle-nodagger.sh= accepts
    file arguments, so we pass all org files to get a full tangle
    (its no-args default only tangles bootstrap files).
  - =run= executes all org-babel blocks via =run-nodagger.sh=.
    The bash blocks call dagger, so nested access is required.
  - =test= runs the test suite via =test-nodagger.sh= (pytest).
    Nested dagger access is required.

  #+begin_src python
  @object_type
  class Dev:
      debian_image: str = "debian:bookworm-slim"

      @function
      def container(self) -> dagger.Container:
          """Dev container with emacs, git, ruff, pytest, and dagger CLI."""
          ctr = (
              dag.container()
              .from_(self.debian_image)
              .with_exec(
                  [
                      "apt-get", "update",
                  ]
              )
              .with_exec(
                  [
                      "apt-get", "install", "-y", "--no-install-recommends",
                      "emacs-nox", "git", "python3", "python3-pip",
                      "python3-venv", "curl", "ca-certificates",
                  ]
              )
              .with_exec(
                  [
                      "pip", "install", "--break-system-packages",
                      "ruff", "pytest", "pytest-asyncio", "pyyaml",
                  ]
              )
              .with_exec(
                  [
                      "sh", "-c",
                      f"curl -fsSL https://dl.dagger.io/dagger/install.sh"
                      f" | DAGGER_VERSION={DAGGER_VERSION} BIN_DIR=/usr/local/bin sh",
                  ]
              )
          )
          # Clone and set up pinned org-mode (cached in this layer)
          ctr = (
              ctr
              .with_exec(
                  [
                      "git", "clone", "--quiet",
                      "https://git.savannah.gnu.org/git/emacs/org-mode.git",
                      "/opt/org",
                  ]
              )
              .with_exec(["git", "-C", "/opt/org", "checkout", "--quiet", ORG_PIN])
              .with_exec(
                  [
                      "emacs", "--batch", "--no-init-file",
                      "--eval",
                      '(progn'
                      '  (push "/opt/org/lisp" load-path)'
                      '  (require (quote autoload))'
                      '  (setq generated-autoload-file "/opt/org/lisp/org-loaddefs.el")'
                      '  (update-directory-autoloads "/opt/org/lisp"))',
                  ]
              )
              .with_exec(
                  [
                      "sh", "-c",
                      "cd /opt/org/lisp && emacs --batch --no-init-file"
                      ' --eval \'(progn'
                      "  (push \\\"/opt/org/lisp\\\" load-path)"
                      "  (load \\\"/opt/org/mk/org-fixup.el\\\")"
                      "  (org-make-org-version"
                      '    \\"0\\" \\"0\\"))\' 2>/dev/null || true',
                  ]
              )
          )
          return ctr

      def _work_container(
          self,
          source: dagger.Directory,
          with_org: bool = False,
      ) -> dagger.Container:
          """Mount source into /work, optionally symlinking pinned org-mode."""
          ctr = (
              self.container()
              .with_workdir("/work")
              .with_directory("/work", source)
          )
          if with_org:
              ctr = ctr.with_exec(
                  [
                      "sh", "-c",
                      "mkdir -p .tangle-deps && ln -sf /opt/org .tangle-deps/org",
                  ]
              )
          return ctr

      @function
      async def tangle(
          self,
          source: Annotated[dagger.Directory, DefaultPath(".")],
      ) -> dagger.Directory:
          """Tangle all org files, returning the directory with generated code."""
          return await (
              self._work_container(source, with_org=True)
              .with_exec(
                  [
                      "sh", "-c",
                      "./tangle-nodagger.sh"
                      " readme.org TECHNICAL.org src/*.org tests/*.org doc/*.org",
                  ]
              )
              .directory("/work")
          )

      @function
      async def run(
          self,
          source: Annotated[dagger.Directory, DefaultPath(".")],
      ) -> dagger.Directory:
          """Execute org-babel blocks and save results."""
          return await (
              self._work_container(source, with_org=True)
              .with_exec(
                  ["./run-nodagger.sh"],
                  experimental_privileged_nesting=True,
              )
              .directory("/work")
          )

      @function
      async def test(
          self,
          source: Annotated[dagger.Directory, DefaultPath(".")],
      ) -> str:
          """Run the test suite."""
          return await (
              self._work_container(source)
              .with_exec(
                  [
                      "pip", "install", "--break-system-packages",
                      "--quiet", "./sdk",
                  ]
              )
              .with_exec(
                  ["./test-nodagger.sh"],
                  experimental_privileged_nesting=True,
              )
              .stdout()
          )
