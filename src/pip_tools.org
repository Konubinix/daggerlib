  Based on the Python version set in [[file:../TECHNICAL.org::*Module entry point][Lib.pip_tools_python_version]],
  with pip-tools installed and a default user. This composes =.from_=,
  =use_user=, and =.with_exec= — the same primitives described in the
  [[file:alpine.org][Alpine]] and [[file:user.org][user]] docs.

  This intentionally uses =python:{version}-alpine= directly instead
  of =alpine_python()=, because =pip-compile= output depends on the
  exact Python version and =alpine_python()= gives whatever Python
  ships with the Alpine release.

  #+NAME: pip-tools

  @function
  def pip_tools(self) -> dagger.Container:
      """Python Alpine with pip-tools and a default user."""
      ctr = dag.container().from_(f"python:{self.pip_tools_python_version}-alpine")
      ctr = self.use_user(ctr)
      return ctr.with_exec(["pip", "--quiet", "install", "pip-tools"])
  #+end_src
